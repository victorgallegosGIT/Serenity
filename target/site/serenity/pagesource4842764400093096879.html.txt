                if (!format_regex.test(strFecha))//La fecha no es válida?
                {
                    elemento.value = "";//Se vacia el textbox

                    return;
                }
                else {
                    var year = parseInt(strFecha.slice(6, 11));//Se obtiene el año

                    if (year > 2199) {//El año es mayor a 2199?
                        elemento.value = "";//Se vacía el textbox
                        return;
                    }
                }
            }
